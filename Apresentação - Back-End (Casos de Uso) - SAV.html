SAV

Sistema de Aluguel de Veículos — Back-End (Casos de Uso)

Tema: Gerenciamento de transações e regras de negócio

O sistema contempla os seguintes processos principais, cada um apresentado por um integrante do grupo:

  • Reserva: o cliente escolhe categoria de veículo, período e agências; o sistema calcula automaticamente os valores financeiros e aplica desconto quando a agência possui histórico operacional comprovado.
  • Check-in: vincula a reserva a um veículo específico; o sistema valida CNH, débitos pendentes e disponibilidade — realizando upgrade automático de categoria quando necessário.
  • Check-out: registra a devolução do veículo; valida quilometragem contra o histórico do veículo e aplica taxa de inspeção a clientes com mais de 3 avarias acumuladas.

Stack: Node.js · Express · Sequelize ORM · PostgreSQL

💻 ReservaService.js (create)

static async create(req) {
  const { dataRetirada, dataDevolucao, clienteId,
          categoriaVeiculoId, seguroId, funcionarioId,
          agenciaRetiradaId, agenciaDevolucaoId } = req.body;
  const erros = [];

  const [cliente, funcionario, categoria] = await Promise.all([
    Cliente.findByPk(clienteId),
    Funcionario.findByPk(funcionarioId),
    CategoriaVeiculo.findByPk(categoriaVeiculoId),
  ]);
  if (!cliente)     erros.push("Cliente não encontrado!");
  if (!funcionario) erros.push("Funcionário não encontrado!");
  if (!categoria)   erros.push("Categoria não encontrada!");

  // Valida regras: data, agências, conflito de período
  await verificarRegrasDeNegocio({ dataRetirada, dataDevolucao,
    clienteId, agenciaRetiradaId, agenciaDevolucaoId, erros });

  // RN1: calcula valores — desconto incluso se agência qualificar
  const { quantidadeDias, valorDiaria, valorSeguro, valorFinal }
    = await calcularValoresFinanceiros({ agenciaRetiradaId,
        categoria, seguro, dataRetirada, dataDevolucao });

  erros.push(...await validarModel(Reserva.build({
    dataRetirada, dataDevolucao, valorDiaria,
    quantidadeDias, valorSeguro, valorFinal })));

  if (erros.length > 0) throw erros.join(" ");

  // INSERT único — sem transação explícita
  const obj = await Reserva.create({
    dataRetirada, dataDevolucao, valorDiaria, quantidadeDias,
    valorSeguro, valorFinal, clienteId, categoriaVeiculoId,
    funcionarioId, seguroId, agenciaRetiradaId, agenciaDevolucaoId,
  });
  return await Reserva.findByPk(obj.id, { include: { all: true } });
}

async function verificarRegrasDeNegocio({
  dataRetirada, dataDevolucao, clienteId,
  agenciaRetiradaId, agenciaDevolucaoId,
  reservaId, erros
}) {
  // Data de retirada não pode ser no passado
  if (dataRetirada)
    validarDataRetirada(dataRetirada, erros);

  // Agências devem existir e estar ativas
  const [agenciaRetirada, agenciaDevolucao] = await Promise.all([
    agenciaRetiradaId ? Agencia.findByPk(agenciaRetiradaId) : null,
    agenciaDevolucaoId ? Agencia.findByPk(agenciaDevolucaoId) : null,
  ]);
  if (agenciaRetiradaId !== undefined)
    validarAgenciaRetirada(agenciaRetirada, erros);
  if (agenciaDevolucaoId !== undefined)
    validarAgenciaDevolucao(agenciaDevolucao, erros);

  // RN2: cliente sem reserva ativa conflitante no período
  if (clienteId && dataRetirada && dataDevolucao)
    await validarConflitoReservaCliente(
      clienteId, dataRetirada, dataDevolucao,
      erros, reservaId ?? null
    );
}

// RN1 — implementada em calcularValoresFinanceiros():
// agencia.reservasRetirada.length >= 2
//   && quantidadeDias >= agencia.limiteDiasDesconto
// → valorFinal *= (1 - percentualDesconto / 100)

Implementação da RN1 🔖 Processo: Reserva

04

RN1 — Desconto por histórico operacional: A agência de retirada aplica percentual de desconto somente se possuir no mínimo 2 reservas Concluídas (histórico comprovado) e a quantidade de dias ≥ limiteDiasDesconto. Ex: agencia1 (limite=3 dias, 10% off) + reserva de 5 dias → R$ 500,00 × 0,90 = R$ 450,00.

-- agencia1: 3 reservas Concluídas (ids 1, 4, 5) → qualifica para desconto

Implementação da RN2 🔖 Processo: Reserva

05

RN2 — Conflito de período: Um cliente não pode ter duas reservas com datas sobrepostas e status ativo (não Cancelada nem Concluída). A sobreposição é detectada pela condição clássica de intervalos: início A < fim B E fim A > início B.

📊 Visualizar Tabelas do Banco de Dados

💻 ReservaService.js — validarConflitoReservaCliente()

async function validarConflitoReservaCliente(
  clienteId, dataRetirada, dataDevolucao, erros, excludeId = null
) {
  const where = {
    clienteId,
    status: { [Op.notIn]: ['Cancelada', 'Concluída'] },
    [Op.and]: [
      { dataRetirada:   { [Op.lt]: dataDevolucao } },  // início da nova < fim da existente
      { dataDevolucao:  { [Op.gt]: dataRetirada  } },  // fim da nova  > início da existente
    ],
  };
  if (excludeId) where.id = { [Op.ne]: excludeId };   // ignora a própria reserva no update

  const conflito = await Reserva.findOne({ where });
  if (conflito)
    erros.push("O cliente já possui uma reserva no período solicitado!");
}

static async create(req) {
  const { dataCheckin, cnhCondutor, cnhValidade,
          quilometragemCheckin, reservaId,
          veiculoId, funcionarioId } = req.body;
  const erros = [];

  const [reserva, funcionario] = await Promise.all([
    Reserva.findByPk(reservaId, { include: { all: true } }),
    Funcionario.findByPk(funcionarioId),
  ]);
  if (!reserva)     erros.push("Reserva não encontrada!");
  if (!funcionario) erros.push("Funcionário não encontrado!");

  let veiculoFinalId = veiculoId;
  if (reserva) {
    veiculoFinalId = await verificarRegrasDeNegocio(
      { cnhCondutor, reserva, veiculoId, isUpdate: false, erros }
    ) ?? veiculoFinalId;
  }
  if (erros.length > 0) throw erros.join(" ");

  return await sequelize.transaction(async (t) => {
    const obj = await Checkin.create({
      dataCheckin, cnhCondutor, cnhValidade, quilometragemCheckin,
      reservaId, veiculoId: veiculoFinalId, funcionarioId
    }, { transaction: t });

    const veiculo = await Veiculo.findByPk(veiculoFinalId, { transaction: t });
    veiculo.status = 'Reservado';
    await veiculo.save({ transaction: t });

    reserva.status = 'Confirmada';
    await reserva.save({ transaction: t });

    return await Checkin.findByPk(obj.id,
      { include: { all: true }, transaction: t });
  }); // COMMIT automático
}

💻 verificarRegrasDeNegocio()

async function verificarRegrasDeNegocio({
  cnhCondutor, reserva, veiculoId, isUpdate, erros
}) {
  if (!isUpdate) {
    // Status da reserva deve ser Pendente
    validarStatusReserva(reserva, erros);
    // RN2: cliente sem débitos pendentes (multas)
    await validarDebitosPendentesCliente(reserva.clienteId, erros);
  }

  // CNH informada deve corresponder ao cliente da reserva
  if (cnhCondutor !== undefined)
    validarCnhCondutor(cnhCondutor, reserva, erros);

  if (!isUpdate) {
    // RN1: resolve veículo disponível na categoria, ou upgrade automático
    const { veiculoFinalId, erros: errosVeiculo }
      = await resolverVeiculoParaCheckin(reserva, veiculoId);
    erros.push(...errosVeiculo);
    return veiculoFinalId;  // veiculoId final (pode ser de categoria superior)
  }
  return veiculoId ?? null;
}

function validarStatusReserva(reserva, erros) {
  if (reserva.status !== 'Pendente')
    erros.push("Só é possível realizar check-in em reservas com status Pendente!");
}

function validarCnhCondutor(cnhCondutor, reserva, erros) {
  if (cnhCondutor !== reserva.cliente.cnh)
    erros.push("A CNH informada não corresponde à CNH cadastrada para o cliente da reserva!");
}

Implementação da RN1 ✅ Processo: Check-in

09

RN1 — Resolução de veículo com upgrade automático: Se não há veículos disponíveis na categoria da reserva, o sistema busca categorias superiores (id > categoriaId) em ordem crescente e seleciona o primeiro veículo disponível. Reserva de catEconômico (id=1) → nenhum disponível → upgrade para catHatch (id=2) → veículo id=3 selecionado.

💻 resolverVeiculoParaCheckin() + buscarVeiculoUpgrade()

async function resolverVeiculoParaCheckin(reserva, veiculoId) {
  // SQL-1: busca veículos disponíveis na categoria da reserva
  const veiculosNaCategoria = await Veiculo.findAll({
    where: { categoriaVeiculoId: reserva.categoriaVeiculoId,
             status: "Disponível" },
  });
  if (veiculosNaCategoria.length === 0) {
    // SQL-2 + SQL-3: upgrade automático
    const veiculoFinalId = await buscarVeiculoUpgrade(reserva.categoriaVeiculoId);
    const erros = veiculoFinalId ? []
      : ["Não há veículos disponíveis e não foi possível realizar upgrade!"];
    return { veiculoFinalId, erros };
  }
  return validarVeiculoSolicitado(reserva, veiculoId);
}

async function buscarVeiculoUpgrade(categoriaVeiculoId) {
  // SQL-2: categorias superiores em ordem crescente de id
  const categoriasSuperiores = await CategoriaVeiculo.findAll({
    where: { id: { [Op.gt]: categoriaVeiculoId } },
    order: [["id", "ASC"]],
  });
  for (const cat of categoriasSuperiores) {
    // SQL-3: primeiro veículo disponível em cada categoria superior
    const veiculo = await Veiculo.findOne({
      where: { categoriaVeiculoId: cat.id, status: "Disponível" }
    });
    if (veiculo) return veiculo.id;
  }
  return null; // nenhum upgrade possível
}

⚙️ SQLs Resultantes (3 passos)

Passo 1: Verifica disponibilidade na categoria da reserva (catEconômico id=1):

SELECT * FROM veiculos
WHERE categoria_veiculo_id = 1
  AND status = 'Disponível';
-- 0 linhas → aciona upgrade

Passo 2: Busca categorias superiores:

WHERE id > 1
ORDER BY id ASC;

LIMIT 1;
-- id=3 (BYD Dolphin) → upgrade realizado!

async function validarDebitosPendentesCliente(clienteId, erros) {
  const debitosPendentes = await Multa.count({
    where: { clienteId, status: "Pendente" }

static async create(req) {
  const { dataCheckout, quilometragemCheckout, nivelCombustivel,
          condicaoPneus, condicaoPalhetas, limpoInternamente,
          limpoExternamente, observacoes,
          checkinId, funcionarioId, avariaIds } = req.body;
  const erros = [];

  const [checkin, funcionario] = await Promise.all([
    Checkin.findByPk(checkinId),
    Funcionario.findByPk(funcionarioId),
  ]);
  if (!checkin)     erros.push("Check-in não encontrado!");
  if (!funcionario) erros.push("Funcionário não encontrado!");

  if (checkin)
    await verificarRegrasDeNegocio({ dataCheckout,
      quilometragemCheckout, obj: checkin, erros });

  if (erros.length > 0) throw erros.join(" ");

  const reserva = await Reserva.findByPk(checkin.reservaId);
  // RN2: calcula taxa de inspeção ANTES de abrir a transação
  const taxaInspecao = await calcularTaxaInspecao(reserva.clienteId);

  return await sequelize.transaction(async (t) => {
    const obj = await Checkout.create({
      dataCheckout, quilometragemCheckout, nivelCombustivel,
      condicaoPneus, condicaoPalhetas, limpoInternamente,
      limpoExternamente, observacoes, checkinId,
      funcionarioId, taxaInspecao
    }, { transaction: t });

    if (avariaIds) await obj.setAvarias(avariaIds, { transaction: t });

    const veiculo = await Veiculo.findByPk(checkin.veiculoId, { transaction: t });
    veiculo.quilometragem = quilometragemCheckout;
    veiculo.status = 'Disponível';
    await veiculo.save({ transaction: t });

    reserva.status = 'Concluída';
    await reserva.save({ transaction: t });

    return await Checkout.findByPk(obj.id,
      { include: { all: true }, transaction: t });
  }); // COMMIT automático
}

async function verificarRegrasDeNegocio({
  dataCheckout, quilometragemCheckout,
  checkinId, obj, erros
}) {
  // Resolve o checkin (update passa checkinId, create passa obj)
  const checkin = checkinId
    ? await Checkin.findByPk(checkinId) : obj;

  if (!checkin) {
    erros.push("Check-in não encontrado!");
    return;
  }

  // RN1: validação de quilometragem em 2 etapas
  if (quilometragemCheckout !== undefined)
    await validarQuilometragem(quilometragemCheckout, checkin, erros);

  // Data de devolução não pode ser anterior ao check-in
  if (dataCheckout !== undefined)
    validarDataCheckout(dataCheckout, checkin, erros);
}

function validarDataCheckout(dataCheckout, checkin, erros) {
  if (new Date(dataCheckout) < new Date(checkin.dataCheckin))
    erros.push("A data de devolução não pode ser anterior à data do check-in!");
}

💻 CheckoutService.js — validarQuilometragem()

async function validarQuilometragem(quilometragemCheckout, checkin, erros) {
  // Etapa 1: deve ser maior que a quilometragem do check-in
  if (parseFloat(quilometragemCheckout) <= parseFloat(checkin.quilometragemCheckin)) {
    erros.push("A quilometragem de devolução deve ser maior que a quilometragem registrada no check-in!");
    return;
  }

  // Etapa 2: não pode ser inferior ao maior valor histórico do veículo
  // Usa raw: true e subQuery: false para geração correta do SQL com MAX + JOIN
  const resultado = await Checkout.findOne({
    attributes: [[sequelize.fn('MAX', sequelize.col('quilometragem_checkout')), 'maxQuilometragem']],
    include: [{ model: Checkin, as: 'checkin', required: true,
                where: { veiculoId: checkin.veiculoId }, attributes: [] }],
    raw: true, subQuery: false,
  });

  const maxQuilometragem = resultado?.maxQuilometragem;
  if (maxQuilometragem != null && parseFloat(quilometragemCheckout) < parseFloat(maxQuilometragem))
    erros.push("A quilometragem de devolução não pode ser inferior à maior quilometragem já registrada para este veículo!");
}

⚙️ SQL Resultante — MAX via INNER JOIN

O subQuery: false força o Sequelize a gerar um JOIN direto com GROUP BY, garantindo que o MAX seja calculado corretamente:

SELECT MAX(checkouts.quilometragem_checkout) AS max_quilometragem
FROM checkouts
INNER JOIN checkins ON checkouts.checkin_id = checkins.id
WHERE checkins.veiculo_id = 5;
-- veiculo_id=5 (Toyota Corolla): checkout3 registrou 31800 km
-- max_quilometragem = 31800 → nova km deve ser ≥ 31800 (e > 50000 do checkin7)

const TAXA_INSPECAO = 150.00;

async function calcularTaxaInspecao(clienteId) {
  // SQL-1: todos os checkins do cliente (via reserva)
  const checkinsDoCliente = await Checkin.findAll({
    include: [{ model: Reserva, as: "reserva",
                required: true, where: { clienteId }, attributes: [] }],
    attributes: ["id"],
  });
  const idsCheckins = checkinsDoCliente.map(c => c.id);
  if (idsCheckins.length === 0) return 0;

  // SQL-2: checkouts desses checkins, com suas avarias
  const checkoutsAnteriores = await Checkout.findAll({
    where: { checkinId: { [Op.in]: idsCheckins } },
    include: [{ model: Avaria, as: "avarias" }],
  });

  // Soma total de avarias via JavaScript
  const totalAvarias = checkoutsAnteriores
    .reduce((total, co) => total + co.avarias.length, 0);

  return totalAvarias > 3 ? TAXA_INSPECAO : 0;
}
